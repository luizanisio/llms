#       sudo apt update && sudo apt install -y wget gnupg ca-certificates
#       wget https://developer.download.nvidia.com/compute/cuda/repos/debian13/x86_64/cuda-keyring_1.1-1_all.deb

# ==============================================================================
#
#   ⚠️  Garantir que o shell enxerga o toolkit antes de compilar:
#       export CUDA_HOME=/usr/local/cuda-12.8
#       export PATH=$CUDA_HOME/bin:$PATH
#       export LD_LIBRARY_PATH=$CUDA_HOME/lib64:${LD_LIBRARY_PATH:-}
#       nvcc --version   # deve mostrar 12.8
#
#   🔬 [A] H100 / servidor (128+ CPUs):
#       TMPDIR=/var/tmp MAX_JOBS=8 pip install flash-attn --no-build-isolation --force-reinstall
#       # ⏱️  ~20 minutos com MAX_JOBS=8
#
#   🏠 [B] RTX 3060 / WSL2 (menos CPUs disponíveis):
#       TMPDIR=/var/tmp MAX_JOBS=4 pip install flash-attn --no-build-isolation --force-reinstall
#       # ⏱️  ~40-60 minutos com MAX_JOBS=4; reduza para MAX_JOBS=2 se a máquina travar
#
#   🔑 Flags importantes:
#   - --no-build-isolation  usa o PyTorch já instalado no env (garante ABI correta)
#   - --force-reinstall     substitui qualquer versão pré-compilada incompatível

# 🆘 SOLUÇÃO DE PROBLEMAS

# ❌ Erro "CUDA_HOME environment variable is not set":
#   → nvcc não está no PATH.
#   → 🔍 Verificar: which nvcc
#   → Se não encontrar: instalar cuda-toolkit-12-8 conforme Passo 0.
#   → Se encontrar: exportar CUDA_HOME, PATH e LD_LIBRARY_PATH antes do pip.
#
# ❌ Erro "undefined symbol: _ZN3c104cuda...":
#   → flash-attn foi compilado contra um CUDA diferente do PyTorch.
#   → 🔧 Solução: garantir CUDA_HOME=/usr/local/cuda-12.8 e repetir o Passo 4.
#
# ❌ Erro "No space left on device":
#   → 🔧 Usar TMPDIR=/var/tmp antes do pip.
